// logger.proto
syntax = "proto3"; 

package logger;

// The central logging service definition
service EventLogger {
  // Method to send a log to the server
  rpc LogEvent (LogRequest) returns (LogResponse); 

  // Method to query logs by node name
  rpc GetNodeLogs (NodeRequest) returns (LogList);
  
  // NEW: Method to query global logs by severity
  rpc GetSeverityLogs (SeverityRequest) returns (LogList);
}

// Request payload for sending a log
message LogRequest {
  string node_name = 1;
  string severity = 2;
  string message = 3;
}

// Response from the server after logging
message LogResponse {
  bool success = 1;
}

// Request payload for querying a specific node
message NodeRequest {
  string node_name = 1;
}

// NEW: Request payload for querying logs by severity
message SeverityRequest {
  string severity = 1;
}

// A single log entry structure
message LogEntry {
  string timestamp = 1;
  string severity = 2;
  string message = 3;
  string node_name = 4;
}

// A list of logs returned to the querying client
message LogList {
  repeated LogEntry logs = 1;
}
